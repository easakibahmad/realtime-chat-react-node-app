# Initialize a new Vite project with React
$ terminal: npm create vite@latest websocket-chat -- --template react

# Navigate into the project directory
$ terminal: cd websocket-chat

# Initialize Node.js project with default settings
$ terminal: npm init -y

# Install required dependencies for Radix UI, Tailwind CSS, and other utilities
$ terminal: npm install @radix-ui/react-select @radix-ui/react-slot class-variance-authority clsx tailwind-merge lucide-react tailwindcss postcss autoprefixer tailwindcss-animate

# Initialize Tailwind CSS with PostCSS and autoprefixer
$ terminal: npx tailwindcss init -p

# Install Node.js type definitions as a development dependency
$ terminal: npm install -D @types/node

# Remove the node_modules folder if you need a fresh installation
$ terminal: rm -rf node_modules
